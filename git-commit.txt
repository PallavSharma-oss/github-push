#!/usr/bin/env sh

echo "Running Python syntax check..."

FILES=$(git diff --cached --name-only --diff-filter=ACM | grep '\.py$')

if [ -n "$FILES" ]; then
  python -m py_compile $FILES
  if [ $? -ne 0 ]; then
    echo "❌ Commit blocked: Python syntax error"
    exit 1
  fi
fi

echo "✅ Syntax OK"
exit 0
